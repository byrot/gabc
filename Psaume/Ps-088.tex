\verset{Misericórdias Dómini * in ætérnum cantábo.}%
%
\verset{In generatiónem et generatiónem * annuntiábo veritátem tuam in ore meo.}%
%
\verset{Quóniam dixísti : In ætérnum misericórdia ædificábitur in cælis : * præparábitur véritas tua in eis.}%
%
\verset{Dispósui testaméntum eléctis meis, iurávi David, servo meo : * Usque in ætérnum præparábo semen tuum.}%
%
\verset{Et ædificábo in generatiónem et generatiónem * sedem tuam.}%
%
\verset{Confitebúntur cæli mirabília tua, Dómine : * étenim veritátem tuam in ecclésia sanctórum.}%
%
\verset{Quóniam quis in núbibus æquábitur Dómino : * símilis erit Deo in fíliis Dei ?}%
%
\verset{Deus, qui glorificátur in consílio sanctórum : * magnus et terríbilis super omnes qui in circúitu eius sunt.}%
%
\verset{Dómine, Deus virtútum, quis símilis tibi ? * potens es, Dómine, et véritas tua in circúitu tuo.}%
%
\verset{Tu domináris potestáti maris : * motum autem flúctuum eius tu mítigas.}%
%
\verset{Tu humiliásti sicut vulnerátum, supérbum : * in brácchio virtútis tuæ dispersísti inimícos tuos.}%
%
\verset{Tui sunt cæli, et tua est terra, orbem terræ et plenitúdinem eius tu fundásti : * aquilónem et mare tu creásti.}%
%
\verset{Thabor et Hermon in nómine tuo exsultábunt : * tuum brácchium cum poténtia.}%
%
\verset{Firmétur manus tua, et exaltétur déxtera tua : * iustítia et iudícium præparátio sedis tuæ.}%
%
\verset{Misericórdia et véritas præcédent fáciem tuam : * beátus pópulus qui scit iubilatiónem.}%
%
\verset{Dómine, in lúmine vultus tui ambulábunt, et in nómine tuo exsultábunt tota die : * et in iustítia tua exaltabúntur.}%
%
\verset{Quóniam glória virtútis eórum tu es : * et in beneplácito tuo exaltábitur cornu nostrum.}%
%
\verset{Quia Dómini est assúmptio nostra, * et Sancti Israël, regis nostri.}%
%
\verset{Tunc locútus es in visióne sanctis tuis et dixísti : * Pósui adiutórium in poténte : et exaltávi eléctum de plebe mea.}%
%
\verset{Invéni David, servum meum : * óleo sancto meo unxi eum.}%
%
\verset{Manus enim mea auxiliábitur ei : * et brácchium meum confortábit eum.}%
%
\verset{Nihil profíciet inimícus in eo, * et fílius iniquitátis non appónet nocére ei.}%
%
\verset{Et concídam a fácie ipsíus inimícos eius : * et odiéntes eum in fugam convértam.}%
%
\verset{Et véritas mea et misericórdia mea cum ipso : * et in nómine meo exaltábitur cornu eius.}%
%
\verset{Et ponam in mari manum eius : * et in flumínibus déxteram eius.}%
%
\verset{Ipse invocábit me : Pater meus es tu : * Deus meus et suscéptor salútis meæ.}%
%
\verset{Et ego primogénitum ponam illum * excélsum præ régibus terræ.}%
%
\verset{In ætérnum servábo illi misericórdiam meam : * et testaméntum meum fidéle ipsi.}%
%
\verset{Et ponam in sǽculum sǽculi semen eius : * et thronum eius sicut dies cæli.}%
%
\verset{Si autem derelíquerint fílii eius legem meam : * et in iudíciis meis non ambuláverint :}%
%
\verset{Si iustítias meas profanáverint : * et mandáta mea non custodíerint :}%
%
\verset{Visitábo in virga iniquitátes eórum : * et in verbéribus peccáta eórum.}%
%
\verset{Misericórdiam autem meam non dispérgam ab eo : * neque nocébo in veritáte mea :}%
%
\verset{Neque profanábo testaméntum meum : * et quæ procédunt de lábiis meis, non fáciam írrita.}%
%
\verset{Semel iurávi in sancto meo : Si David méntiar : * semen eius in ætérnum manébit.}%
%
\verset{Et thronus eius sicut sol in conspéctu meo, * et sicut luna perfécta in ætérnum : et testis in cælo fidélis.}%
%
\verset{Tu vero repulísti et despexísti : * distulísti Christum tuum.}%
%
\verset{Evertísti testaméntum servi tui : * profanásti in terra Sanctuárium eius.}%
%
\verset{Destruxísti omnes sepes eius : * posuísti firmaméntum eius formídinem.}%
%
\verset{Diripuérunt eum omnes transeúntes viam : * factus est oppróbrium vicínis suis.}%
%
\verset{Exaltásti déxteram depriméntium eum : * lætificásti omnes inimícos eius.}%
%
\verset{Avertísti adiutórium gládii eius : * et non es auxiliátus ei in bello.}%
%
\verset{Destruxísti eum ab emundatióne : * et sedem eius in terram collisísti.}%
%
\verset{Minorásti dies témporis eius : * perfudísti eum confusióne.}%
%
\verset{Usquequo, Dómine, avértis in finem : * exardéscet sicut ignis ira tua ?}%
%
\verset{Memoráre quæ mea substántia : * numquid enim vane constituísti omnes fílios hóminum ?}%
%
\verset{Quis est homo, qui vivet et non vidébit mortem : * éruet ánimam suam de manu ínferi ?}%
%
\verset{Ubi sunt misericórdiæ tuæ antíquæ, Dómine, * sicut iurásti David in veritáte tua ?}%
%
\verset{Memor esto, Dómine, oppróbrii servórum tuórum * (quod contínui in sinu meo) multárum géntium.}%
%
\verset{Quod exprobravérunt inimíci tui, Dómine, * quod exprobravérunt commutatiónem Christi tui.}%
%
\verset{Benedíctus Dóminus in ætérnum : * fiat, fiat.}
