%% Ps. 112
\verset{Louez le Seigneur, vous ses serviteurs ; louez le nom du Seigneur.}
\verset{Que le nom du Seigneur soit béni, dès maintenant et dans tous les siècles.}
\verset{Du lever du soleil à son son couchant, le nom du Seigneur est digne de louange.}
\verset{Le Seigneur est élevé au-dessus de toutes les nations, et sa gloire est au-dessus des cieux.}
\verset{Qui est semblable au Seigneur notre Dieu, qui habite dans les hauteurs, et qui regarde ce qui est humble  au ciel et sur la terre ?}
\verset{Il tire l'indigent de la poussière, et relève le pauvre du fumier,}
\verset{Pour le placer avec les princes, avec les princes de son peuple.}
\verset{Il fait habiter celle qui était stérile dans la maison, comme une mère joyeuse au milieux de ses enfants.}
